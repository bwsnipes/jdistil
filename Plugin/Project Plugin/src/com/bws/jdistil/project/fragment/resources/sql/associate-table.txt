create table ASSOCIATE-TABLE-NAME
(
   PARENT-ID-COLUMN-NAME    INTEGER		NOT NULL,
   DEPENDENT-ID-COLUMN-NAME   INTEGER		NOT NULL
)
;

alter table ASSOCIATE-TABLE-NAME add constraint pk_ASSOCIATE-TABLE-NAME
   primary key (PARENT-ID-COLUMN-NAME, DEPENDENT-ID-COLUMN-NAME)
;

alter table ASSOCIATE-TABLE-NAME add constraint fk_ASSOCIATE-TABLE-NAME_1
   foreign key (PARENT-ID-COLUMN-NAME) references PARENT-TABLE-NAME (PARENT-ID-COLUMN-NAME)
;

alter table ASSOCIATE-TABLE-NAME add constraint fk_ASSOCIATE-TABLE-NAME_2
   foreign key (DEPENDENT-ID-COLUMN-NAME) references DEPENDENT-TABLE-NAME (DEPENDENT-ID-COLUMN-NAME)
;
